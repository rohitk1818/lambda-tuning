var kvmName = context.getVariable("request.queryparam.kvmName");
if (!kvmName) {
    throw new Error("KVM name not provided");
}

// Use apigee.access.get to retrieve key from dynamic KVM
apigee.access.get(kvmName, 'auth', function (error, value) {
    if (error) {
        throw new Error("KVM fetch error: " + error);
    } else {
        context.setVariable("private.var_header", value);
    }
});
